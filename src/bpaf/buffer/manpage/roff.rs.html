27/// You can concatenate multiple `Roff` documents with `+` or `+=`.
28#[derive(Debug, Default, Clone)]
29pub(crate) struct Roff {
30    payload: FreeMonoid<Escape>,
31    /// keep or strip newlines from inserted text
32    pub strip_newlines: bool,
33}
34
35/// Font selector
36#[derive(Debug, Clone, Copy, Eq, PartialEq)]
37pub(crate) enum Font {
38    /// Roman font - regular font that should be used for most of the text
39    Roman,
40
41    /// Bold font
42    Bold,
43
44    /// Italic font
45    Italic,
46}
47
48/// Escape code used to return to the previous font
49pub(crate) const RESTORE_FONT: &str = "\\fP";
50
51impl Font {
52    /// Escape sequence needed to set this font, None for default font
53    ///
54    pub(crate) fn escape(self) -> &'static str {
55        match self {
56            Font::Bold => "\\fB",
57            Font::Italic => "\\fI",
58            Font::Roman => "\\fR",
59        }
60    }
61}
62
63impl Roff {
64    /// Create new raw Roff document
65    #[must_use]
66    pub(crate) fn new() -> Self {
67        Self::default()
68    }
69
70    /// Chainable setter for `strip_newlines` field
71    ///
72    /// `strip_newlines` specifies if [`render`](Self::render) should keep all the newline characters
73    /// inside added text newlines can come with a special meaning, for example adding a section
74    /// header relies, newlines are automatically stripped from [`control`](Self::control) arguments.
75    pub(crate) fn strip_newlines(&mut self, state: bool) -> &mut Self {
76        self.strip_newlines = state;
77        self
78    }
79
80    /// Insert a raw control sequence
81    ///
82    /// `name` should not contain initial `'.'`.
83    /// Arguments are taken from an iterator and escaped accordingly
84    pub(crate) fn control<S, I>(&mut self, name: &str, args: I) -> &mut Self
85    where
86        S: AsRef<str>,
87        I: IntoIterator<Item = S>,
88    {
89        self.payload.push_str(Escape::UnescapedAtNewline, ".");
90        self.payload.push_str(Escape::Unescaped, name);
91        for arg in args {
92            // empty macro argument can be specified as "", mostly useful for TH macro
93            // which takes several arguments positionally
94            let mut s = arg.as_ref();
95            if s.is_empty() {
96                s = "\"\"";
97            }
98            self.payload
99                .push_str(Escape::Unescaped, " ")
100                .push_str(Escape::Spaces, s);
101        }
102        self.payload.push_str(Escape::UnescapedAtNewline, "");
103        self
104    }
105
106    /// A variant of control that takes no parameters
107    pub(crate) fn control0(&mut self, name: &str) -> &mut Self {
108        self.payload.push_str(Escape::UnescapedAtNewline, ".");
109        self.payload.push_str(Escape::Unescaped, name);
110        self.payload.push_str(Escape::UnescapedAtNewline, "");
111        self
112    }
113
114    /// Insert a line break in the Roff document source
115    ///
116    /// This will not show up in the output of the roff program.
117    pub(crate) fn roff_linebreak(&mut self) -> &mut Self {
118        self.payload.push_str(Escape::UnescapedAtNewline, "");
119        self
120    }
121
122    /// Insert raw escape sequence
123    ///
124    /// You can use all the notations for the escapes, they will be copied into the output stream
125    /// as is without extra checks or escapes.
126    pub(crate) fn escape(&mut self, arg: &str) -> &mut Self {
127        self.payload.push_str(Escape::Unescaped, arg);
128        self
129    }
130
131    /// Insert a plain text string, special characters are escaped
132    ///
133    pub(crate) fn plaintext(&mut self, text: &str) -> &mut Self {
134        if self.strip_newlines {
135            self.payload.push_str(Escape::SpecialNoNewline, text);
136        } else {
137            self.payload.push_str(Escape::Special, text);
138        }
139        self
140    }
141
142    /// Insert one or more string slices using custom font for each one
143    pub(crate) fn text(&mut self, text: &[(Font, &str)]) -> &mut Self {
144        let mut prev_font = None;
145        for (font, item) in text {
146            if prev_font == Some(font) {
147                self.plaintext(item.as_ref());
148            } else {
149                let escape = font.escape();
150                self.escape(escape).plaintext(item.as_ref());
151                prev_font = Some(font);
152            }
153        }
154        if prev_font.is_some() {
155            self.escape(RESTORE_FONT);
156        }
157        self
158    }
159
160    /// Render Roff document to `String`
161    ///
162    /// This method creates a valid ROFF document which can be fed to a ROFF implementation
163    #[must_use]
164    pub(crate) fn render(&self, ap: Apostrophes) -> String {
165        let mut res = Vec::with_capacity(self.payload.payload_size() * 2);
166        if ap == Apostrophes::Handle {
167            res.extend(super::escape::APOSTROPHE_PREABMLE.as_bytes());
168        }
169        super::escape::escape(&self.payload, &mut res, ap);
170        String::from_utf8(res).expect("Should be valid utf8 by construction")
171    }
172}

174impl AddAssign<&Roff> for Roff {
175    fn add_assign(&mut self, rhs: &Roff) {
176        self.payload += &rhs.payload;
177        self.strip_newlines = rhs.strip_newlines;
178    }
179}
180
181impl Add<&Roff> for Roff {
182    type Output = Self;
183
184    fn add(mut self, rhs: &Roff) -> Self::Output {
185        self += rhs;
186        self
187    }
188}
189
190impl<'a> Extend<&'a Roff> for Roff {
191    fn extend<I: IntoIterator<Item = &'a Roff>>(&mut self, iter: I) {
192        for i in iter {
193            *self += i;
194        }
195    }
196}

199mod test {
200    use super::{Apostrophes, Font, Roff};
201    const NO_AP: Apostrophes = Apostrophes::DontHandle;
202
203    #[test]
204    fn escape_dash_in_plaintext() {
205        let text = Roff::default().plaintext("-").render(NO_AP);
206        assert_eq!(r"\-", text);
207    }
208
209    #[test]
210    fn escape_backslash_in_plaintext() {
211        let text = Roff::default().plaintext(r"\x").render(NO_AP);
212        assert_eq!(r"\\x", text);
213    }
214
215    #[test]
216    fn escape_backslash_and_dash_in_plaintext() {
217        let text = Roff::default().plaintext(r"\-").render(NO_AP);
218        assert_eq!(r"\\\-", text);
219    }
220
221    #[test]
222    fn escapes_leading_control_chars_and_space_in_plaintext() {
223        let text = Roff::default()
224            .plaintext("foo\n.bar\n'yo\n hmm")
225            .render(NO_AP);
226        assert_eq!("foo\n\\&.bar\n\\&'yo\n hmm", text);
227    }
