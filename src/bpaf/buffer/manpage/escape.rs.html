1//! Escape and concatenate string slices according to Roff Escape rules
2
3/// Apostrophes handling configuration
4///
5/// To generate manpages you most likely want to have this in `Handle` state.
6/// See <https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=507673#65> for more details
7#[derive(Eq, PartialEq, Copy, Clone)]
8#[allow(dead_code)] // it is used in the test
9pub enum Apostrophes {
10    /// Replace apostrophes with special code that
11    Handle,
12    /// Leave apostrophes as is
13    DontHandle,
14}
15
16/// Replacement for apostrophes, if active
17///
18/// <https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=507673#65>
19const APOSTROPHE: &str = "\\*(Aq";
20
21#[allow(clippy::doc_markdown)]
22/// A preamble added to the start of rendered output.
23///
24/// This defines a string variable that contains an apostrophe. For
25/// historical reasons, there seems to be no other portable way to
26/// represent apostrophes across various implementations of the ROFF
27/// language. In implementations that produce output like PostScript
28/// or PDF, an apostrophe gets typeset as a right single quote, which
29/// looks different from an apostrophe. For terminal output ("ASCII"),
30/// such as when using nroff, an apostrophe looks indistinguishable
31/// from a right single quote. For manual pages, and similar content,
32/// an apostrophe is more generally desired than the right single
33/// quote, so we convert all apostrophe characters in input text into
34/// a use of the string variable defined in the preamble.
35///
36/// Used when apostrophe handle is enabled.
37///
38/// See: <https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=507673#65>
39pub(crate) const APOSTROPHE_PREABMLE: &str = r#".ie \n(.g .ds Aq \(aq
40.el .ds Aq '
41"#;
42
43/// Escaping rules
44#[derive(Debug, Copy, Clone, Eq, PartialEq)]
45pub(crate) enum Escape {
46    /// Insert a newline character unless on a new line already
47    UnescapedAtNewline,
48
49    /// Escape space characters (`' '`) and newlines (`'\n'`)
50    /// This escape is used for control sequence arguments
51    Spaces,
52
53    /// Escape characters roff considers special:
54    /// - `' '`, `'.'` and `'\''` at the beginning of the line,
55    /// - `'-'` and `'\\'` when inside the body
56    /// - replace `'\''` with APOSTROPHE if Apostrophes handling is enabled
57    Special,
58
59    /// Similar to [`Special`](Escape::Special) but also replaces `'\n'` with `' '`
60    SpecialNoNewline,
61
62    /// Input is written as is
63    Unescaped,
64}
65
66#[cfg(test)]
67/// Escape a sequence of string slices according to escaping rules and store results to `String`
68///
69/// See also [`escape`] if it is desired to reuse existing storage capacity
70pub(crate) fn escape_to_string<'a, I>(items: I, ap: Apostrophes) -> String
71where
72    I: IntoIterator<Item = (&'a Escape, &'a str)>,
73{
74    let mut res = Vec::new();
75    escape(items, &mut res, ap);
76    String::from_utf8(res).expect("Output should be utf8 by construction")
77}
78
79/// Escape a sequence of string slices according to escaping rules
80///
81/// Writes results to `out`, result should be a valid utf8 string as long as `out` starts empty or
82/// contains a valid utf8 sequence
83pub(crate) fn escape<'a, I>(items: I, out: &mut Vec<u8>, ap: Apostrophes)
84where
85    I: IntoIterator<Item = (&'a Escape, &'a str)>,
86{
87    let mut at_line_start = true;
88    for (&meta, payload) in items {
89        if !at_line_start && meta == Escape::UnescapedAtNewline {
90            out.push(b'\n');
91            at_line_start = true;
92        }
93        for &c in payload.as_bytes() {
94            match meta {
95                Escape::Spaces => {
96                    if c == b' ' || c == b'\n' {
97                        out.extend_from_slice(b"\\ ");
98                    } else {
99                        out.push(c);
100                    }
101                }
102                Escape::Special | Escape::SpecialNoNewline => {
103                    if at_line_start && (c == b'.' || c == b'\'') {
104                        out.extend_from_slice(b"\\&");
105                    }
106                    if c == b'\\' || c == b'-' {
107                        out.push(b'\\');
108                    }
109                    if ap == Apostrophes::Handle && c == b'\'' {
110                        out.extend_from_slice(APOSTROPHE.as_bytes());
111                    } else if meta == Escape::SpecialNoNewline && c == b'\n' {
112                        out.push(b' ');
113                        at_line_start = false;
114                        continue;
115                    } else {
116                        out.push(c);
117                    }
118                }
119                Escape::Unescaped | Escape::UnescapedAtNewline => {
120                    out.push(c);
121                }
122            }
123            at_line_start = c == b'\n';
124        }
125    }
126}

128#[cfg(test)]
129mod test {
130    use super::{escape_to_string, Apostrophes, Escape};
131
132    #[test]
133    fn sample() {
134        let ap = Apostrophes::Handle;
135        let items: &[(Escape, &str)] = &[
136            (Escape::Unescaped, "\\fI"),
137            (Escape::Special, "test"),
138            (Escape::Unescaped, "\\fP"),
139        ];
140        let output = escape_to_string(items.iter().map(|p| (&p.0, p.1)), ap);
141        assert_eq!("\\fItest\\fP", output);
142    }
