use bpaf::batteries::*;

#[derive(Debug, Copy, Clone, Eq, PartialEq, Ord, PartialOrd)]
enum Level {
   Error,
   Warning,
   Info,
   Debug,
   Trace,
}

fn verbose() -> impl Parser<Level> {
    use Level::*;
    verbose_by_slice(2, [Error, Warning, Info, Debug, Trace])
}
